void rotateLogFiles(const char* logDirectory) {
    DIR* dir = opendir(logDirectory);
    if (dir == NULL) {
        printf("Failed to open log directory\n");
        return;
    }

    struct dirent* entry;
    struct stat fileStat;
    char filePath[256];
    int fileCount = 0;

    // Count the number of log files
    while ((entry = readdir(dir)) != NULL) {
        snprintf(filePath, sizeof(filePath), "%s/%s", logDirectory, entry->d_name);
        if (stat(filePath, &fileStat) == 0 && S_ISREG(fileStat.st_mode)) {
            fileCount++;
        }
    }

    // Close and reopen the directory to reset the position
    closedir(dir);
    dir = opendir(logDirectory);

    // Delete excess log files
    while (fileCount > MAX_LOG_FILES && (entry = readdir(dir)) != NULL) {
        snprintf(filePath, sizeof(filePath), "%s/%s", logDirectory, entry->d_name);
        if (stat(filePath, &fileStat) == 0 && S_ISREG(fileStat.st_mode)) {
            remove(filePath);
            fileCount--;
        }
    }

    closedir(dir);
}

To mitigate this and allow the CPU to scale down appropriately, you should ensure that your main loop allows for some idling. If you’re using a polling mechanism for buttons, consider implementing one of the following strategies:

Use Interrupts: Instead of polling, use GPIO interrupts to respond to button presses. This will free your main loop from continually checking the buttons and allow the CPU to enter light sleep.

Add Delays: If you must use polling, add a delay (like vTaskDelay) in your polling loop to give the CPU time to idle. This will allow the power management system to reduce the frequency or enter light sleep when possible.

Task Yielding: If you’re in a FreeRTOS task and there’s no work left to do for a while, you can call vTaskDelay or taskYIELD() to allow other tasks or the idle task to run, which can trigger the CPU to scale down.

Enabling and correctly configuring the power management features of the ESP32 can significantly affect the power consumption, but you must ensure that your application is compatible with it. void save_config_item(const char *key, int value) {
    char line[MAX_LINE_LENGTH] = {0};
    char temp_buffer[MAX_LINE_LENGTH * CONFIG_FILE_LINES] = {0};
    char new_line[MAX_LINE_LENGTH] = {0};

    // Convert value to string
    snprintf(new_line, MAX_LINE_LENGTH, "%s = %d\n", key, value);

    // Open for reading
    FILE *file = fopen(Config_Path, "r");
    bool key_found = false;
    
    if (file) {
        char read_key[MAX_LINE_LENGTH];
        char *temp_ptr = temp_buffer; // Use a pointer to track position in temp_buffer

        // memset(read_key, 0, sizeof(read_key)); // Clean the buffer
        // Read each line
        while (fgets(line, sizeof(line), file)) {
            // Check if the line contains the key
            if (sscanf(line, "%[^=] =", read_key) == 1 && strcmp(read_key, key) == 0) {
                ESP_LOGI(TAG3, "KEY FOUND.");
                key_found = true; // Key has been found
                // Add updated key-value pair, no need to copy this line to temp_buffer
                strcpy(temp_ptr, new_line);
                temp_ptr += strlen(new_line); // Move pointer past new line
                continue; // Skip appending the current line
            }
            // Copy other lines to temp_buffer
            strcpy(temp_ptr, line);
            temp_ptr += strlen(line); // Move pointer
        }
        fclose(file); // Close the file after reading contents

        // Append the new key-value pair if the key was not found
        if (!key_found) {
            ESP_LOGI(TAG3, "Appending new key-value pair.");
            strcpy(temp_ptr, new_line);
        }
    }

    // Re-open the file for writing (this will clear the file if it exists)
    file = fopen(Config_Path, "w");
    if (!file) {
        ESP_LOGE(TAG3, "Error opening configuration file for writing.");
        return;
    }

    // Write updated buffer back to file
    fwrite(temp_buffer, sizeof(char), strlen(temp_buffer), file);
    fclose(file);
    ESP_LOGI(TAG3, "Config item saved: %s", new_line);
}

void draw_start_attack(u8g2_t *u8g2, uint8_t Attack_Timeout) {
    // Initialize your display code here
    u8g2_ClearBuffer(u8g2);
    u8g2_SetDrawColor(u8g2, 1);
    u8g2_SetFont(u8g2, u8g2_font_profont22_tr);
    u8g2_DrawStr(u8g2, 16, 38, "ATTACKING");
    u8g2_SendBuffer(u8g2);

    // Reset your event group
    xEventGroupClearBits(attack_event_group, ATTACK_COUNTER_UPDATE ATTACK_COMPLETE);

    // Start attack in a separate task
    xTaskCreate(attack_task, "attack_task", 2048, NULL, 5, NULL);

    // Countdown timer
    for (int counter = Attack_Timeout; counter >= 0; counter--) {
        // Wait for the update event
        xEventGroupWaitBits(attack_event_group, ATTACK_COUNTER_UPDATE, pdTRUE, pdFALSE, portMAX_DELAY);

        // Draw the current timer value here, i.e., `counter` / `Attack_Timeout`
        char timer_str[20];
        snprintf(timer_str, 20, "%d / %d", counter, Attack_Timeout);
        u8g2_DrawStr(u8g2, 16, 58, timer_str); // Place this properly
        u8g2_SendBuffer(u8g2);

        // Check if the attack has completed
        if(xEventGroupGetBits(attack_event_group) & ATTACK_COMPLETE) {
            break; // Exit the loop if the attack is complete
        }
    }

    // Now the attack has finished, save the result
    hccapx_t *hccapx_data = hccapx_serializer_get();
    // Use your function for saving hccapx_data to SD card or handling it as needed
    save_hccapx_to_sd("/sdcard/attack_result.hccapx", hccapx_data);
}
